# External libraries
find_package(Eigen3 REQUIRED NO_MODULE)
include(libigl)
igl_include(core)

add_library(modcam_objlib OBJECT "${MODCAM_SRC}" "${MODCAM_HEADERS}")
target_link_libraries(
	modcam_objlib
	PUBLIC
		Eigen3::Eigen
	PRIVATE
		doctest
		igl::core
)
target_include_directories(
	modcam_objlib
	PUBLIC
		$<BUILD_INTERFACE:${MODCAM_PROJECT_DIR}/include>
		$<INSTALL_INTERFACE:include>
)
target_compile_options(
	modcam_objlib
	PRIVATE
		$<$<CXX_COMPILER_ID:MSVC>:/W4 /WX>
		$<$<NOT:$<CXX_COMPILER_ID:MSVC>>:-Wall -Wextra -Wpedantic>
)
set_target_properties(modcam_objlib PROPERTIES POSITION_INDEPENDENT_CODE ON)

if(MODCAM_BUILD_CPP)
	add_library(modcam_cpp SHARED $<TARGET_OBJECTS:modcam_objlib>)
	set_target_properties(
		modcam_cpp
		PROPERTIES
			LIBRARY_OUTPUT_DIRECTORY "${PROJECT_BINARY_DIR}"
	)
endif()

if(MODCAM_BUILD_PYTHON)
	include("py_modcam/SourceList.cmake")
	pybind11_add_module(py_modcam "${PY_MODCAM_SRC}")
	target_link_libraries(
		py_modcam
		PRIVATE
			modcam_objlib
			Eigen3::Eigen
	)
	set_target_properties(
		py_modcam
		PROPERTIES
			LIBRARY_OUTPUT_NAME "modcam"
			LIBRARY_OUTPUT_DIRECTORY "${PROJECT_BINARY_DIR}"
	)
endif()
